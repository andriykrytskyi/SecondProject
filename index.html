const canvas = document.getElementById('gameCanvas'); const ctx = canvas.getContext('2d'); const grid = 20; // size of each cell const canvasSize = canvas.width; let snake = [{x: 8 * grid, y: 8 * grid}]; let direction = {x: 0, y: 0}; let food = {x: Math.floor(Math.random() * (canvasSize/grid)) * grid, y: Math.floor(Math.random() * (canvasSize/grid)) * grid}; let score = 0; function gameLoop() { // move snake const head = {x: snake[0].x + direction.x*grid, y: snake[0].y + direction.y*grid}; // game over conditions if (head.x < 0 || head.x >= canvasSize || head.y < 0 || head.y >= canvasSize || snakeCollision(head)) { alert("Game Over! Score: " + score); snake = [{x: 8*grid, y: 8*grid}]; direction = {x: 0, y: 0}; score = 0; food = {x: Math.floor(Math.random() * (canvasSize/grid)) * grid, y: Math.floor(Math.random() * (canvasSize/grid)) * grid}; } snake.unshift(head); // eat food if (head.x === food.x && head.y === food.y) { score++; food = {x: Math.floor(Math.random() * (canvasSize/grid)) * grid, y: Math.floor(Math.random() * (canvasSize/grid)) * grid}; } else { snake.pop(); } draw(); } function snakeCollision(head) { for (let i = 1; i < snake.length; i++) { if (head.x === snake[i].x && head.y === snake[i].y) return true; } return false; } function draw() { ctx.fillStyle = "#222"; ctx.fillRect(0, 0, canvasSize, canvasSize); // draw snake ctx.fillStyle = "#0f0"; snake.forEach(segment => ctx.fillRect(segment.x, segment.y, grid-2, grid-2)); // draw food ctx.fillStyle = "#f00"; ctx.fillRect(food.x, food.y, grid-2, grid-2); // draw score ctx.fillStyle = "#fff"; ctx.font = "20px Arial"; ctx.fillText("Score: " + score, 10, 20); } // control snake document.addEventListener('keydown', e => { switch(e.key) { case 'ArrowUp': if (direction.y === 0) direction = {x:0, y:-1}; break; case 'ArrowDown': if (direction.y === 0) direction = {x:0, y:1}; break; case 'ArrowLeft': if (direction.x === 0) direction = {x:-1, y:0}; break; case 'ArrowRight': if (direction.x === 0) direction = {x:1, y:0}; break; } }); // start game setInterval(gameLoop, 100);
